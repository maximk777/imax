Секция 3: Модель данных

SQLite schema — минимум для MVP

-- Локальный профиль пользователя
CREATE TABLE identity (
    id            INTEGER PRIMARY KEY CHECK (id = 1),  -- singleton
    seed_phrase   TEXT NOT NULL,          -- BIP39 encrypted at rest
    public_key    BLOB NOT NULL,          -- Ed25519 public key
    nickname      TEXT NOT NULL,          -- отображаемое имя
    created_at    INTEGER NOT NULL        -- unix timestamp
);

-- Контакты (известные пиры)
CREATE TABLE contacts (
    public_key    BLOB PRIMARY KEY,       -- Ed25519 peer public key
    nickname      TEXT NOT NULL,          -- ник пира
    node_id       BLOB,                  -- iroh NodeId для reconnect
    added_at      INTEGER NOT NULL,      -- unix timestamp
    last_seen     INTEGER                -- последний онлайн
);

-- Чаты (1-to-1 для MVP)
CREATE TABLE chats (
    id            TEXT PRIMARY KEY,       -- UUID
    peer_key      BLOB NOT NULL UNIQUE,   -- FK → contacts.public_key
    created_at    INTEGER NOT NULL,
    last_message  INTEGER,               -- FK → messages.id (для превью)
    unread_count  INTEGER DEFAULT 0
);

-- Сообщения
CREATE TABLE messages (
    id            TEXT PRIMARY KEY,       -- UUID (генерируется отправителем)
    chat_id       TEXT NOT NULL,          -- FK → chats.id
    sender_key    BLOB NOT NULL,          -- кто отправил
    content       BLOB NOT NULL,          -- зашифрованный текст (E2E)
    plaintext     TEXT,                   -- расшифрованный (только локально)
    status        TEXT NOT NULL           -- pending | sent | delivered | read
                  CHECK (status IN ('pending','sent','delivered','read')),
    created_at    INTEGER NOT NULL,       -- timestamp отправки
    received_at   INTEGER                 -- timestamp получения
);

-- Ожидающие invite-коды
CREATE TABLE pending_invites (
    code          TEXT PRIMARY KEY,       -- invite код
    created_at    INTEGER NOT NULL,
    expires_at    INTEGER                 -- TTL
);

-- Индексы
CREATE INDEX idx_messages_chat ON messages(chat_id, created_at);
CREATE INDEX idx_messages_status ON messages(status) WHERE status = 'pending';

Ключевые решения

  • identity — singleton: один профиль на устройство, CHECK (id = 1)
  • content (BLOB) + plaintext (TEXT): храним и шифрованное (для пересылки), и расшифрованное (для поиска/отображения)
  • status flow: pending → sent → delivered → read
  • pending_invites: TTL для автоочистки просроченных кодов
  • UUID для messages: генерируется отправителем, уникален глобально — нет конфликтов между пирами
